1. 安装依赖:
   pip install -r requirements.txt

2. 测试新功能:
   python sgs_parser.py
   python structure_validator.py
   python test_modules.py

3. 下载真实数据集 (需要API密钥):
   export MP_API_KEY="your_api_key"
   python mp_dataset_loader.py --dataset mp20

4. 训练模型:
   python run_crystalicl.py \
       --model_name Qwen/Qwen3-8B \
       --data_path ./data/mp20.json \
       --do_train \
       --num_epochs 3

5. 完整评估:
   python evaluate_complete.py \
       --model_path ./crystalicl_qwen3_8b_output \
       --test_data ./data/test.json \
       --num_samples 1000
